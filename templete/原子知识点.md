<%*
// 1. 获取当前文件名作为默认值
let title = tp.file.title;

// 2. 弹窗询问标题 (如果直接按回车，就保留原文件名)
const input = await tp.system.prompt("请输入笔记标题：", title);

// 3. 如果输入了新名字，且不为空，则重命名文件
if (input && input !== title) {
    await tp.file.rename(input);
    title = input; // 更新 title 变量用于下方标题显示
}
%>
